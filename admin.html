Supabase Database Setup Instructions

If using Supabase, copy and execute this SQL snippet in the Supabase **SQL Editor** to create the necessary tables:

-- 1. Projects Table
create table if not exists projects (
  id uuid default gen_random_uuid() primary key,
  title text not null,
  description text,
  category text,
  year text,
  link text,
  tags text[],
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- Enable RLS and add public policies
alter table projects enable row level security;
create policy "Public Read" on projects for select using (true);
create policy "Admin Write" on projects for all using (true); -- Set appropriate auth rules as needed

-- 2. Testimonials Table
create table if not exists testimonials (
  id uuid default gen_random_uuid() primary key,
  client_name text not null,
  company_role text,
  review_text text,
  image_url text,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

alter table testimonials enable row level security;
create policy "Public Read" on testimonials for select using (true);
create policy "Admin Write" on testimonials for all using (true);

-- 3. About Me Table
create table if not exists about_me (
  key text primary key,
  value jsonb,
  updated_at timestamp with time zone default timezone('utc'::text, now()) not null
);

alter table about_me enable row level security;
create policy "Public Read" on about_me for select using (true);
create policy "Admin Write" on about_me for all using (true);

-- 4. Blogs Table
create table if not exists blogs (
  id uuid default gen_random_uuid() primary key,
  title text not null,
  content text not null,
  excerpt text,
  status text default 'Published', -- Published, Draft
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

alter table blogs enable row level security;
create policy "Public Read" on blogs for select using (true);
create policy "Admin Write" on blogs for all using (true);
